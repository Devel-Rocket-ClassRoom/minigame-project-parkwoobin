using UnityEngine;

public abstract class ItemBase : MonoBehaviour, IInteractable
{
    [SerializeField] protected int   expReward   = 10;
    [SerializeField] protected int   scoreReward = 10;
    [SerializeField] protected float feverReward = 5f;

    bool _collected;

    public virtual void Interact(PlayerController player)
    {
        if (_collected) return;
        _collected = true;
        OnCollect(player);
        Destroy(gameObject);
    }

    protected abstract void OnCollect(PlayerController player);

    protected void GrantBaseRewards(PlayerController player)
    {
        player.GetComponent<ExpManager>()?.AddExp(expReward);
        ScoreManager.Instance?.AddScore(scoreReward);
        player.GetComponent<PlayerStats>()?.AddFever(feverReward);
    }
}
