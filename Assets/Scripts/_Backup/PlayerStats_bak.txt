using UnityEngine;
using UnityEngine.Events;

public class PlayerStats : MonoBehaviour
{
    [Header("Health")]
    [SerializeField] float maxHealth = 100f;
    float _health;

    [Header("Hunger")]
    [SerializeField] float maxHunger = 100f;
    [SerializeField] float hungerDecayRate = 2f;
    float _hunger;

    [Header("Experience")]
    [SerializeField] int expToNextStage = 100;
    int _exp;

    [Header("Fever")]
    [SerializeField] float maxFever = 100f;
    float _fever;

    [Header("Money")]
    int _money;

    public float Health   => _health;
    public float Hunger   => _hunger;
    public int   Exp      => _exp;
    public float Fever    => _fever;
    public int   Money    => _money;

    public float HealthRatio => _health / maxHealth;
    public float HungerRatio => _hunger / maxHunger;
    public float ExpRatio    => (float)_exp / expToNextStage;
    public float FeverRatio  => _fever / maxFever;

    public static UnityAction<float> OnHealthChanged;
    public static UnityAction<float> OnHungerChanged;
    public static UnityAction<int>   OnExpChanged;
    public static UnityAction<float> OnFeverChanged;
    public static UnityAction<int>   OnMoneyChanged;
    public static UnityAction        OnDeath;
    public static UnityAction        OnExpFull;

    void Start()
    {
        _health = maxHealth;
        _hunger = maxHunger;
        _exp    = 0;
        _fever  = 0;
        _money  = 0;
    }

    void Update()
    {
        if (!GameManager.Instance.IsPlaying) return;
        ModifyHunger(-hungerDecayRate * Time.deltaTime);
    }

    public void ModifyHealth(float delta)
    {
        _health = Mathf.Clamp(_health + delta, 0f, maxHealth);
        OnHealthChanged?.Invoke(HealthRatio);
        if (_health <= 0f) OnDeath?.Invoke();
    }

    public void ModifyHunger(float delta)
    {
        _hunger = Mathf.Clamp(_hunger + delta, 0f, maxHunger);
        OnHungerChanged?.Invoke(HungerRatio);
        if (_hunger <= 0f) ModifyHealth(-10f * Time.deltaTime);
    }

    public void AddExp(int amount)
    {
        _exp = Mathf.Min(_exp + amount, expToNextStage);
        OnExpChanged?.Invoke(_exp);
        if (_exp >= expToNextStage) OnExpFull?.Invoke();
    }

    public void AddFever(float amount)
    {
        _fever = Mathf.Clamp(_fever + amount, 0f, maxFever);
        OnFeverChanged?.Invoke(FeverRatio);
    }

    public void ConsumeFever()
    {
        _fever = 0f;
        OnFeverChanged?.Invoke(0f);
    }

    public void AddMoney(int amount)
    {
        _money += amount;
        OnMoneyChanged?.Invoke(_money);
        SaveManager.Instance?.AddMoney(amount);
    }

    public void SetMaxHealth(float value)       => maxHealth = value;
    public void SetHungerDecayRate(float value) => hungerDecayRate = value;
}
