using UnityEngine;

public class ItemSpawner : MonoBehaviour
{
    [SerializeField] GameObject[] foodPrefabs;
    [SerializeField] GameObject[] moneyPrefabs;
    [SerializeField] Transform[]  spawnPoints;
    [SerializeField] int foodCount  = 5;
    [SerializeField] int moneyCount = 3;

    void Start() { SpawnItems(); }

    void SpawnItems()
    {
        for (int i = 0; i < foodCount  && foodPrefabs.Length  > 0; i++) SpawnAt(foodPrefabs[Random.Range(0, foodPrefabs.Length)]);
        for (int i = 0; i < moneyCount && moneyPrefabs.Length > 0; i++) SpawnAt(moneyPrefabs[Random.Range(0, moneyPrefabs.Length)]);
    }

    void SpawnAt(GameObject prefab)
    {
        if (spawnPoints.Length == 0) return;
        Transform point = spawnPoints[Random.Range(0, spawnPoints.Length)];
        Instantiate(prefab, point.position, Quaternion.identity, transform);
    }
}
