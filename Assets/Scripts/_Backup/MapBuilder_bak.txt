using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Tilemaps;

public class MapBuilder : MonoBehaviour
{
    [Header("타일맵 레이어")]
    [SerializeField] Tilemap groundTilemap;
    [SerializeField] Tilemap decorTilemap;

    [Header("맵 데이터")]
    [SerializeField] MapData[] stageMapDatas;

    [Header("스폰 프리팹")]
    [SerializeField] GameObject playerPrefab;
    [SerializeField] GameObject personEnemyPrefab;
    [SerializeField] GameObject catEnemyPrefab;
    [SerializeField] GameObject foodItemPrefab;
    [SerializeField] GameObject tutorialSignPrefab;
    [SerializeField] GameObject stageGoalPrefab;

    [Header("배경")]
    [SerializeField] ParallaxBackground parallax;
    [SerializeField] SpriteRenderer     skyRenderer;
    [SerializeField] SpriteRenderer     midRenderer;
    [SerializeField] SpriteRenderer     nearRenderer;

    readonly List<GameObject> _spawnedObjects = new List<GameObject>();

    void OnEnable()  => StageManager.OnStageLoaded += OnStageLoaded;
    void OnDisable() => StageManager.OnStageLoaded -= OnStageLoaded;

    void Start()
    {
        int idx = StageManager.Instance != null ? StageManager.Instance.CurrentStageIndex : 0;
        Build(idx);
    }

    void OnStageLoaded(int index) => Build(index);

    public void Build(int stageIndex)
    {
        if (stageIndex < 0 || stageIndex >= stageMapDatas.Length)
        {
            Debug.LogWarning($"[MapBuilder] stageIndex {stageIndex} 범위 초과");
            return;
        }
        MapData data = stageMapDatas[stageIndex];
        if (data == null) return;

        ClearAll();
        PlaceTiles(data);
        SetupBackgrounds(data);
        SpawnObjects(data);
        SetCameraBounds(data);

        Debug.Log($"[MapBuilder] '{data.mapName}' 빌드 완료");
    }

    void PlaceTiles(MapData data)
    {
        if (data.rows == null || data.rows.Length == 0) return;
        int rowCount = data.rows.Length;

        for (int rowIdx = 0; rowIdx < rowCount; rowIdx++)
        {
            int y = (rowCount - 1 - rowIdx) + data.originOffset.y;
            string row = data.rows[rowIdx];

            for (int x = 0; x < row.Length; x++)
            {
                char c = row[x];
                TileBase tile = CharToTile(c, data);
                if (tile == null) continue;

                var pos = new Vector3Int(x + data.originOffset.x, y, 0);
                bool isDecor = (c == 'F');
                Tilemap target = isDecor ? decorTilemap : groundTilemap;
                if (target != null) target.SetTile(pos, tile);
            }
        }
    }

    TileBase CharToTile(char c, MapData d) => c switch
    {
        'G' => d.groundTile, 'U' => d.undergroundTile, 'W' => d.wallTile,
        'B' => d.boxTile,    'F' => d.fenceTile,        'P' => d.platformTile,
        _   => null
    };

    void SetupBackgrounds(MapData data)
    {
        if (skyRenderer  != null && data.skyBackground  != null) skyRenderer.sprite  = data.skyBackground;
        if (midRenderer  != null && data.midBackground  != null) midRenderer.sprite  = data.midBackground;
        if (nearRenderer != null && data.nearBackground != null) nearRenderer.sprite = data.nearBackground;
    }

    void SpawnObjects(MapData data)
    {
        foreach (var sp in data.spawnPoints)
        {
            GameObject prefab = GetPrefab(sp.type);
            if (prefab == null) continue;
            var go = Instantiate(prefab, sp.worldPos, Quaternion.identity);
            go.name = $"{sp.type}_{sp.label}";
            if (sp.type == SpawnType.PersonEnemy && sp.param > 0f)
            {
                var enemy = go.GetComponent<PersonEnemy>();
                if (enemy != null) enemy.PatrolRadius = sp.param;
            }
            _spawnedObjects.Add(go);
        }
    }

    GameObject GetPrefab(SpawnType type) => type switch
    {
        SpawnType.Player       => playerPrefab,
        SpawnType.PersonEnemy  => personEnemyPrefab,
        SpawnType.CatEnemy     => catEnemyPrefab,
        SpawnType.FoodItem     => foodItemPrefab,
        SpawnType.TutorialSign => tutorialSignPrefab,
        SpawnType.StageGoal    => stageGoalPrefab,
        _                      => null
    };

    void SetCameraBounds(MapData data)
    {
        var cam = FindObjectOfType<CameraFollow>();
        if (cam == null) return;
        cam.SetBounds(data.leftBound, data.rightBound);
    }

    void ClearAll()
    {
        groundTilemap?.ClearAllTiles();
        decorTilemap?.ClearAllTiles();
        foreach (var go in _spawnedObjects)
            if (go != null) Destroy(go);
        _spawnedObjects.Clear();
    }
}
