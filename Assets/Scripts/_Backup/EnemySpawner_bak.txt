using UnityEngine;

public class EnemySpawner : MonoBehaviour
{
    [Header("Prefabs")]
    [SerializeField] GameObject[] personPrefabs;
    [SerializeField] GameObject[] catPrefabs;
    [SerializeField] GameObject[] dogPrefabs;

    [Header("Spawn Points")]
    [SerializeField] Transform[] spawnPoints;

    [Header("Config from Stage")]
    [SerializeField] int personCount = 2;
    [SerializeField] int catCount    = 1;
    [SerializeField] int dogCount    = 0;

    void Start()
    {
        SpawnEnemies();
    }

    void SpawnEnemies()
    {
        SpawnGroup(personPrefabs, personCount);
        SpawnGroup(catPrefabs,    catCount);
        SpawnGroup(dogPrefabs,    dogCount);
    }

    void SpawnGroup(GameObject[] prefabs, int count)
    {
        if (prefabs == null || prefabs.Length == 0) return;
        for (int i = 0; i < count; i++)
        {
            Transform point = spawnPoints[Random.Range(0, spawnPoints.Length)];
            Instantiate(prefabs[Random.Range(0, prefabs.Length)],
                        point.position, Quaternion.identity, transform);
        }
    }
}
