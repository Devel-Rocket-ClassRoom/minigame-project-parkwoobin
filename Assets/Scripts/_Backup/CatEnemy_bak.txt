using UnityEngine;

public class CatEnemy : EnemyBase
{
    [SerializeField] bool  isBoss    = false;
    [SerializeField] float territory = 3f;

    Vector2 _home;

    protected override void Awake()
    {
        base.Awake();
        _home = transform.position;
        if (isBoss) { damage *= 2f; attackRange *= 1.5f; detectionRange *= 1.5f; }
    }

    protected override void UpdateBehaviour(float distToPlayer)
    {
        switch (State)
        {
            case EnemyState.Patrol:
                if (isBoss) break;
                Rb.linearVelocity = Vector2.zero;
                break;

            case EnemyState.Chase:
                float homeDist = Vector2.Distance(transform.position, _home);
                if (!isBoss && homeDist > territory)
                {
                    OnLosePlayer();
                    MoveTowards(_home, chaseSpeed);
                    return;
                }
                MoveTowards(Player.position, chaseSpeed);
                TryAttackPlayer(distToPlayer);
                break;
        }
    }

    public void TakeDamage(float dmg)
    {
        // TODO: health for cat enemies in 2nd version
        Debug.Log($"Cat took {dmg} damage");
    }

    protected override void OnLosePlayer()
    {
        base.OnLosePlayer();
        ScoreManager.Instance?.AddScore(ScoreManager.WIN_CAT_FIGHT);
    }
}
