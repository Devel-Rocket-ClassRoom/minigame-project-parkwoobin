using UnityEngine;

public class FoodItem : ItemBase
{
    public enum FoodType { Fish, Tuna, Milk, ChickenLeg, RottenFood }
    [SerializeField] FoodType foodType = FoodType.Fish;

    protected override void OnCollect(PlayerController player)
    {
        var hunger = player.GetComponent<HungerSystem>();
        var health = player.GetComponent<HealthSystem>();

        switch (foodType)
        {
            case FoodType.Fish:        hunger?.Feed(25f);  scoreReward = ScoreManager.FISH; break;
            case FoodType.Tuna:        hunger?.Feed(50f);  scoreReward = ScoreManager.TUNA; break;
            case FoodType.Milk:        health?.Heal(15f);  scoreReward = 20; break;
            case FoodType.ChickenLeg:  hunger?.Feed(35f);  scoreReward = 80; break;
            case FoodType.RottenFood:  hunger?.Feed(20f);  health?.TakeDamage(10f); scoreReward = 10; break;
        }
        GrantBaseRewards(player);
    }
}
