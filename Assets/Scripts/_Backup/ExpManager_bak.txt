using UnityEngine;

public class ExpManager : MonoBehaviour
{
    PlayerStats _stats;

    void Awake() { _stats = GetComponent<PlayerStats>(); }

    void Start()    { PlayerStats.OnExpFull += HandleExpFull; }
    void OnDestroy(){ PlayerStats.OnExpFull -= HandleExpFull; }

    public void AddExp(int amount)
    {
        _stats.AddExp(amount);
        _stats.AddFever(amount * 0.5f);
    }

    void HandleExpFull() { StageManager.Instance?.CompleteStage(); }
}
