using UnityEngine;

public class PersonEnemy : EnemyBase
{
    public enum PersonType { Normal, Child, ShopOwner, Guard, Kind }

    [SerializeField] PersonType personType = PersonType.Normal;

    Vector2 _patrolStart;
    Vector2 _patrolEnd;
    bool    _patrolRight = true;

    [SerializeField] float patrolRange = 3f;

    /// <summary>MapBuilder에서 스폰 후 순찰 반경을 주입할 때 사용</summary>
    public float PatrolRadius
    {
        get => patrolRange;
        set
        {
            patrolRange = value;
            _patrolEnd  = _patrolStart + Vector2.right * patrolRange;
        }
    }

    protected override void Awake()
    {
        base.Awake();
        _patrolStart = transform.position;
        _patrolEnd   = _patrolStart + Vector2.right * patrolRange;
        ApplyTypeConfig();
    }

    void ApplyTypeConfig()
    {
        switch (personType)
        {
            case PersonType.Guard:      detectionRange *= 1.8f; chaseSpeed *= 1.2f; break;
            case PersonType.ShopOwner:  chaseSpeed *= 1.4f; break;
            case PersonType.Child:      patrolSpeed *= 1.5f; detectionRange *= 0.7f; break;
            case PersonType.Kind:       detectionRange = 0f; break;
        }
    }

    protected override void UpdateBehaviour(float distToPlayer)
    {
        if (personType == PersonType.Kind) { TryGiveFood(distToPlayer); return; }

        switch (State)
        {
            case EnemyState.Patrol: Patrol(); break;
            case EnemyState.Chase:
                MoveTowards(Player.position, chaseSpeed);
                TryAttackPlayer(distToPlayer);
                break;
        }
    }

    void Patrol()
    {
        Vector2 target = _patrolRight ? _patrolEnd : _patrolStart;
        MoveTowards(target, patrolSpeed);
        if (Vector2.Distance(transform.position, target) < 0.2f)
            _patrolRight = !_patrolRight;
    }

    void TryGiveFood(float dist) { /* TODO */ }

    protected override void OnDetectPlayer()
    {
        base.OnDetectPlayer();
        ScoreManager.Instance?.AddScore(0);
    }

    protected override void OnLosePlayer()
    {
        base.OnLosePlayer();
        ScoreManager.Instance?.AddScore(ScoreManager.EVADE_PERSON);
    }
}
