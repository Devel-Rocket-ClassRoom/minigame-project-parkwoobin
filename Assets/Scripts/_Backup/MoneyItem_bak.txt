using UnityEngine;

public class MoneyItem : ItemBase
{
    public enum MoneyType { Coin, Bill, Wallet, DeliveryBag }
    [SerializeField] MoneyType moneyType = MoneyType.Coin;

    protected override void OnCollect(PlayerController player)
    {
        var stats = player.GetComponent<PlayerStats>();

        int amount = moneyType switch
        {
            MoneyType.Coin        => 10,
            MoneyType.Bill        => 50,
            MoneyType.Wallet      => FeverManager.Instance?.IsFeverActive == true ? 200 : 80,
            MoneyType.DeliveryBag => Random.Range(30, 150),
            _                     => 10
        };

        stats?.AddMoney(amount);
        scoreReward = ScoreManager.MONEY;
        GrantBaseRewards(player);
    }
}
